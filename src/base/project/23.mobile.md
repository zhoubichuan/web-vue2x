# 六.其他功能（移动端）

::: tip 前言
主要介绍移动端打包
:::

::: demo

```vue
<template>
  <div>
    <el-form
      ref="loginForm"
      :model="form"
      :rules="rules"
    >
      <h3 class="login-title">csvtojson</h3>
      <el-form-item prop="task">
        <el-input
          type="text"
          placeholder="请输入账号"
          v-model="form.task"
        />
      </el-form-item>
      <el-form-item>
        <web-button type="primary" v-on:click="onSubmit('loginForm')">
          转 换
        </web-button>
      </el-form-item>
    </el-form>
  </div>
</template>

<script>
export default {
  name: "Login",
  data() {
    return {
      form: {
        task: "/usr/src/zhoubichuan/prod/web-nginx/config/aaa.csv",
      },
      rules: {
        task: [
          { required: true, message: "账号不可为空", trigger: "blur" },
        ],
      },
    }
  },
  methods: {
    onSubmit(form) {
      this.$refs[form].validate( async (valid) => {
        if (valid) {
         const res = await this.$api.example.csvtojson(this.form)
         if(res.code === 200){
            this.$message.success("成功!")
         }
        } else {
          return false
        }
      })
    },
  },
}
</script>

<style>

}

.login-title {
  text-align: center;
  margin: 0 auto 40px auto;
  color: #303133;
}
</style>
```
